ctrl + alt + t

pwd (print working directory)

cd (change directory)

ls (exibir conteudo)

cd .. (voltar uma pasta)

cd ../.. (voltar duas pastas)

clear(limpar)

mkdir (make directory)

touch texto.txt (criar arquivo de texto chamado "texto")

echo "Olá, mundo!" > texto2.txt (criar um arquivo de texto "texto2" que recebe "Olá, mundo!")

cat texto2.txt (retorna o conteudo do arquivo de texto)

mv texto2.txt olamundo.txt (mover)

cat olamundo.txt (validar a execução)

rmdir (remove directory)

&& (permite encadear comandos)
